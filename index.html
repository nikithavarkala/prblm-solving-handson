Find the Smaller Angle

        var Minimal_Angle = (h, m) => 
        {
          let hr= (h*360)/12+(m*360)/(12*60);
          let min=(m*360)/60;
          let angle=Math.abs(hr-min)
          if(angle>180){
            return 360-angle;
          }
          else{
            return angle;
          }
        };

Perfect Number Check.

        var Perfect_Check = (N) => 
        {
          if(N===1){
            return "YES";
          }
          let m=N;
          let sum=0;
          for(let i=0;i< N;i++){
            if(N%i===0){
              sum+=i;
            }
          }  
          if(sum===m){
            return "YES";
          }      
          else{
            return "NO";
          }
        };

Reverse a Number.

        var Reverse_Number = (N) => 
        {
          let res=0;
          while(N>0){
            r=N%10;
            res=res*10+r;
            N=Math.floor(N/10);
          }
          return res;
        };
